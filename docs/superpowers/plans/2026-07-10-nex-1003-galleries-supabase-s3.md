# NEX-1003: Galleries Table with Supabase S3 Upload Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a `galleries` table and a Clerk-protected dashboard page where a user uploads, views, edits, and deletes gallery entries backed by images in Supabase Storage.

**Architecture:** New Drizzle table (`gallerySchema`) scoped per `clerkUserId`. A server-side `SupabaseStorage` wrapper (service-role key, never exposed to the client) handles upload/delete of image objects. Server actions co-located under the new dashboard route call the storage wrapper and the DB. A client form (`react-hook-form` + zod) drives create/update; a server-rendered list drives read/delete.

**Tech Stack:** Next.js App Router, DrizzleORM (PostgreSQL), `@supabase/supabase-js`, Zod, react-hook-form, Clerk, next-intl, Vitest.

## Global Constraints

- Env vars only via `src/libs/Env.ts` (`@t3-oss/env-nextjs` + Zod) — never `process.env` directly elsewhere.
- Named exports only, no default exports (except Next.js pages).
- Absolute imports via `@/` unless same directory.
- No `useMemo`/`useCallback`; avoid `useEffect`.
- Single `props` param with inline type, accessed as `props.foo`.
- All user-visible strings via `getTranslations` (server) / `useTranslations` (client); keys added to `src/locales/en.json` and `src/locales/th.json`.
- Image upload constraints: max 5MB, MIME in `image/jpeg | image/png | image/webp`. No server-side resize/compress.
- `remove()` must delete the storage object before the DB row; abort on storage-delete failure to avoid orphaned rows.
- Package manager is `bun` (`bun.lock` present) — use `bun add` / `bun run`.
- Locales in this repo are `en` and `th` (not `fr` — `src/locales/fr.json` does not exist).

```ts
import { Env } from '@/libs/Env';

const supabase = createClient(Env.SUPABASE_URL, Env.SUPABASE_SERVICE_ROLE_KEY);

export const uploadGalleryImage = async (file: File, key: string): Promise<void> => {
  const { error } = await supabase.storage
    .from(Env.SUPABASE_STORAGE_BUCKET)
    .upload(key, file, { upsert: false });

  if (error) {
    throw new Error(error.message);
  }
};

export const deleteGalleryImage = async (key: string): Promise<void> => {
  const { error } = await supabase.storage
    .from(Env.SUPABASE_STORAGE_BUCKET)
    .remove([key]);

  if (error) {
    throw new Error(error.message);
  }
};

export const getPublicUrl = (key: string): string => {
  const { data } = supabase.storage.from(Env.SUPABASE_STORAGE_BUCKET).getPublicUrl(key);
  return data.publicUrl;
};
```

```ts
import * as z from 'zod';

export const GALLERY_MAX_IMAGE_BYTES = 5 * 1024 * 1024;
export const GALLERY_ALLOWED_MIME_TYPES = ['image/jpeg', 'image/png', 'image/webp'] as const;

export const GalleryValidation = z.object({
  title: z.string().min(1).max(120),
  image: z
    .instanceof(File)
    .refine((file) => file.size <= GALLERY_MAX_IMAGE_BYTES, 'File too large')
    .refine(
      (file) => GALLERY_ALLOWED_MIME_TYPES.includes(file.type as (typeof GALLERY_ALLOWED_MIME_TYPES)[number]),
      'Unsupported file type',
    ),
});
```

```ts
'use server';

import { auth } from '@clerk/nextjs/server';
import { eq } from 'drizzle-orm';
import { revalidatePath } from 'next/cache';
import { db } from '@/libs/DB';
import { deleteGalleryImage, getPublicUrl, uploadGalleryImage } from '@/libs/SupabaseStorage';
import { gallerySchema } from '@/models/Schema';
import { GalleryValidation } from '@/validations/GalleryValidation';

export type GalleryActionState = {
  status: 'idle' | 'success' | 'error';
};

const buildImageKey = (clerkUserId: string, fileName: string): string => {
  const extension = fileName.split('.').pop() ?? 'jpg';
  return `galleries/${clerkUserId}/${crypto.randomUUID()}.${extension}`;
};

export const listGalleries = async (clerkUserId: string) => {
  return db
    .select()
    .from(gallerySchema)
    .where(eq(gallerySchema.clerkUserId, clerkUserId));
};

export const createGallery = async (
  _prevState: GalleryActionState,
  formData: FormData,
): Promise<GalleryActionState> => {
  const { userId } = await auth();

  if (!userId) {
    return { status: 'error' };
  }

  const parsed = GalleryValidation.safeParse({
    title: formData.get('title'),
    image: formData.get('image'),
  });

  if (!parsed.success) {
    return { status: 'error' };
  }

  const imagePath = buildImageKey(userId, parsed.data.image.name);
  await uploadGalleryImage(parsed.data.image, imagePath);
  const imageUrl = getPublicUrl(imagePath);

  await db.insert(gallerySchema).values({
    clerkUserId: userId,
    title: parsed.data.title,
    imagePath,
    imageUrl,
  });

  revalidatePath('/dashboard/gallery');

  return { status: 'success' };
};

export const updateGallery = async (
  _prevState: GalleryActionState,
  formData: FormData,
): Promise<GalleryActionState> => {
  const { userId } = await auth();
  const id = formData.get('id');

  if (!userId || typeof id !== 'string') {
    return { status: 'error' };
  }

  const image = formData.get('image');
  const parsed = GalleryValidation.pick({ title: true }).safeParse({
    title: formData.get('title'),
  });

  if (!parsed.success) {
    return { status: 'error' };
  }

  const [existing] = await db
    .select()
    .from(gallerySchema)
    .where(eq(gallerySchema.id, id));

  if (!existing || existing.clerkUserId !== userId) {
    return { status: 'error' };
  }

  let imagePath = existing.imagePath;
  let imageUrl = existing.imageUrl;

  if (image instanceof File && image.size > 0) {
    const imageParsed = GalleryValidation.pick({ image: true }).safeParse({ image });
    if (!imageParsed.success) {
      return { status: 'error' };
    }
    imagePath = buildImageKey(userId, image.name);
    await uploadGalleryImage(image, imagePath);
    imageUrl = getPublicUrl(imagePath);
    await deleteGalleryImage(existing.imagePath);
  }

  await db
    .update(gallerySchema)
    .set({ title: parsed.data.title, imagePath, imageUrl })
    .where(eq(gallerySchema.id, id));

  revalidatePath('/dashboard/gallery');

  return { status: 'success' };
};

export const deleteGallery = async (id: string): Promise<GalleryActionState> => {
  const { userId } = await auth();

  if (!userId) {
    return { status: 'error' };
  }

  const [existing] = await db
    .select()
    .from(gallerySchema)
    .where(eq(gallerySchema.id, id));

  if (!existing || existing.clerkUserId !== userId) {
    return { status: 'error' };
  }

  try {
    await deleteGalleryImage(existing.imagePath);
  } catch {
    return { status: 'error' };
  }

  await db.delete(gallerySchema).where(eq(gallerySchema.id, id));

  revalidatePath('/dashboard/gallery');

  return { status: 'success' };
};
```

- [ ] **Step 7: Build the page**

Create `src/app/[locale]/(auth)/dashboard/gallery/page.tsx`:

```tsx
import type { Metadata } from 'next';
import { auth } from '@clerk/nextjs/server';
import { getTranslations, setRequestLocale } from 'next-intl/server';
import { redirect } from 'next/navigation';
import { listGalleries } from './actions';
import { GalleryForm } from '@/components/gallery/GalleryForm';
import { GalleryList } from '@/components/gallery/GalleryList';

export async function generateMetadata(props: {
  params: Promise<{ locale: string }>;
}): Promise<Metadata> {
  const { locale } = await props.params;
  const t = await getTranslations({ locale, namespace: 'GalleryPage' });

  return {
    title: t('meta_title'),
    description: t('meta_description'),
  };
}

export default async function GalleryPage(props: {
  params: Promise<{ locale: string }>;
}) {
  const { locale } = await props.params;
  setRequestLocale(locale);

  const { userId } = await auth();
  if (!userId) {
    redirect('/sign-in');
  }

  const galleries = await listGalleries(userId);

  return (
    <div className="space-y-6">
      <h1 className="text-3xl font-bold tracking-tight text-gray-900 dark:text-gray-100">
        {(await getTranslations({ locale, namespace: 'GalleryPage' }))('meta_title')}
      </h1>
      <GalleryForm />
      <GalleryList items={galleries} />
    </div>
  );
}
```
